import streamlit as st
import pandas as pd
import random
import io

# 페이지 설정
st.set_page_config(page_title="학생 자리 배치 도구", layout="wide")

st.title("🪑 학생 무작위 자리 배치 시스템")
st.markdown("학생 명단을 입력하고 '배치 시작' 버튼을 눌러주세요.")

# 사이드바 설정
st.sidebar.header("설정")
num_cols = st.sidebar.number_input("강의실 가로 좌석 수 (열)", min_value=1, max_value=10, value=5)

# 학생 명단 입력
student_input = st.text_area("학생 이름을 입력하세요 (쉼표 또는 엔터로 구분)", 
                            placeholder="김철수, 이영희, 박지민...", height=200)

if st.button("🎲 자리 배치 시작"):
    # 이름 파싱 (공백 제거 및 빈 값 제외)
    students = [s.strip() for s in student_input.replace('\n', ',').split(',') if s.strip()]
    
    if not students:
        st.error("학생 이름을 입력해주세요!")
    else:
        # 1. 무작위 섞기
        random.shuffle(students)
        
        # 2. 좌석 배열 생성
        total_students = len(students)
        rows = (total_students // num_cols) + (1 if total_students % num_cols != 0 else 0)
        
        # 부족한 자리는 '빈 자리'로 채움
        padded_students = students + ["(빈 자리)"] * (rows * num_cols - total_students)
        
        # 2차원 배열(표) 형태로 변환
        seating_chart = [padded_students[i:i + num_cols] for i in range(0, len(padded_students), num_cols)]
        
        # 데이터프레임 생성
        df = pd.DataFrame(seating_chart, columns=[f"{i+1}열" for i in range(num_cols)])
        df.index = [f"{i+1}행" for i in range(rows)]

        # 3. 결과 출력
        st.subheader("📍 배치 결과")
        st.table(df) # 깔끔한 표 형태로 출력

        # 4. 엑셀 다운로드 기능
        output = io.BytesIO()
        with pd.ExcelWriter(output, engine='openpyxl') as writer:
            df.to_excel(writer, index=True, sheet_name='Seat_Chart')
        
        processed_data = output.getvalue()
        
        st.download_button(
            label="📥 배치 결과 엑셀로 다운로드",
            data=processed_data,
            file_name="seat_assignment.xlsx",
            mime="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"
        )
        st.success(f"총 {total_students}명의 배치가 완료되었습니다!")
